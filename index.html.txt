import React, { useState, useEffect, useRef } from 'react';
import { Trophy, Zap, MousePointer2, Download, Play, ShieldAlert, Target, Activity, ZapOff, CheckCircle2 } from 'lucide-react';

/**
 * PavCA Assessment App: Spatial Visual-Motor Coordination Task
 * VERSION 5.5 - Increased Horizontal Spacing for Robust Target Calculation
 */

const App = () => {
  // --- State Management ---
  const [phase, setPhase] = useState('INSTRUCTIONS'); 
  const [trial, setTrial] = useState(0);
  const [subStep, setSubStep] = useState('IDLE'); // IDLE, CUE, REWARD_AVAILABLE
  const [score, setScore] = useState(0);
  const [sessionData, setSessionData] = useState([]);
  const [chargeLevel, setChargeLevel] = useState(0);
  const [showCue, setShowCue] = useState(false);
  const [isAnchored, setIsAnchored] = useState(false);
  const [isFastBlinking, setIsFastBlinking] = useState(false);

  // --- Tracking Refs ---
  const trialStartTime = useRef(0);
  const currentTrialMetrics = useRef({});
  const containerRef = useRef(null);
  const trialActiveRef = useRef(false);

  // --- Configuration ---
  const TOTAL_ACQ_TRIALS = 10; 
  const PIT_DURATION = 30000; 
  const EXTINCTION_TRIALS = 5;
  
  // Variable reward timing logic (8-12s, Avg 10s)
  const getRandomCueDuration = () => Math.floor(Math.random() * (12000 - 8000 + 1)) + 8000;

  // --- PIT Charge Decay Logic ---
  useEffect(() => {
    let decayTimer;
    if (phase === 'PIT') {
      decayTimer = setInterval(() => {
        setChargeLevel(prev => Math.max(0, prev - 2));
      }, 150);
    }
    return () => clearInterval(decayTimer);
  }, [phase]);

  // --- Logic: Data Logging ---
  const startTracking = () => {
    trialStartTime.current = performance.now();
    currentTrialMetrics.current = {
      firstMoveLatency: null,
      firstMoveAngle: null,
      firstMoveTarget: null, 
      cueDwellTime: 0,
      goalDwellTime: 0,
      peakVelocity: 0,
      interactionCount: 0,
      lastTimestamp: performance.now(),
      lastX: null,
      lastY: null,
      actualDuration: 0
    };
  };

  const handleMouseMove = (e) => {
    if (!containerRef.current) return;
    const rect = containerRef.current.getBoundingClientRect();
    const x = e.clientX - rect.left;
    const y = e.clientY - rect.top;
    const now = performance.now();

    const centerX = rect.width / 2;
    const centerY = rect.height - 110; 
    const distFromAnchor = Math.sqrt(Math.pow(x - centerX, 2) + Math.pow(y - centerY, 2));
    
    // Anchor detection
    if (distFromAnchor < 45) {
      if (!isAnchored) setIsAnchored(true);
    } else {
      if (isAnchored) setIsAnchored(false);
    }

    if (subStep === 'IDLE' && phase !== 'PIT') return;
    
    const dt = now - (currentTrialMetrics.current.lastTimestamp || now);

    // Latency & Target Tracking
    // Increased threshold to 80px to detect movement out of anchor more robustly
    if (currentTrialMetrics.current.firstMoveLatency === null && distFromAnchor > 80) {
      currentTrialMetrics.current.firstMoveLatency = now - trialStartTime.current;
      
      // Categorize target based on movement direction
      // Since objects are further apart, we use a slightly wider center dead-zone
      if (x < centerX - 70) {
        currentTrialMetrics.current.firstMoveTarget = "GOAL";
      } else if (x > centerX + 70) {
        currentTrialMetrics.current.firstMoveTarget = "CUE";
      } else {
        currentTrialMetrics.current.firstMoveTarget = "CENTER";
      }
    }

    // Vigor Tracking
    if (dt > 0 && currentTrialMetrics.current.lastX !== null) {
      const dist = Math.sqrt(Math.pow(x - currentTrialMetrics.current.lastX, 2) + Math.pow(y - currentTrialMetrics.current.lastY, 2));
      const velocity = dist / dt;
      if (velocity > currentTrialMetrics.current.peakVelocity) {
        currentTrialMetrics.current.peakVelocity = velocity;
      }
    }

    // Dwell Time (using 40% boundaries for far-left/far-right)
    if (x > rect.width * 0.6) currentTrialMetrics.current.cueDwellTime += dt;
    if (x < rect.width * 0.4) currentTrialMetrics.current.goalDwellTime += dt;

    currentTrialMetrics.current.lastTimestamp = now;
    currentTrialMetrics.current.lastX = x;
    currentTrialMetrics.current.lastY = y;
  };

  const finalizeTrialData = (phaseName, tNum, actualDuration) => {
    const totalTime = (currentTrialMetrics.current.cueDwellTime || 0) + (currentTrialMetrics.current.goalDwellTime || 0);
    const metrics = {
      phase: phaseName,
      trial: tNum + 1,
      duration: actualDuration ? (actualDuration / 1000).toFixed(1) + 's' : 'N/A',
      latency: currentTrialMetrics.current.firstMoveLatency ? Math.round(currentTrialMetrics.current.firstMoveLatency) + 'ms' : 'N/A',
      vigor: (currentTrialMetrics.current.peakVelocity || 0).toFixed(3),
      target: currentTrialMetrics.current.firstMoveTarget || "N/A",
      signBias: totalTime > 0 ? (currentTrialMetrics.current.cueDwellTime / totalTime).toFixed(3) : 0,
      intensity: currentTrialMetrics.current.interactionCount || 0
    };
    setSessionData(prev => [...prev, metrics]);
  };

  useEffect(() => {
    if ((phase === 'ACQUISITION' || phase === 'EXTINCTION') && subStep === 'IDLE' && isAnchored && !trialActiveRef.current) {
        runTrialCycle();
    }
  }, [isAnchored, phase, subStep]);

  const runTrialCycle = async () => {
    trialActiveRef.current = true;
    startTracking();
    setSubStep('CUE');
    setIsFastBlinking(false);

    const totalDuration = getRandomCueDuration();
    const preBlinkDuration = totalDuration - 3000;

    await new Promise(r => setTimeout(r, preBlinkDuration));
    
    if (trialActiveRef.current) {
        setIsFastBlinking(true);
        await new Promise(r => setTimeout(r, 3000));
    }

    if (trialActiveRef.current) {
        setIsFastBlinking(false);
        if (phase === 'ACQUISITION') {
            setSubStep('REWARD_AVAILABLE');
            currentTrialMetrics.current.actualDuration = totalDuration;
        } else {
            finalizeTrialData(phase, trial, totalDuration);
            completeTrial();
        }
    }
  };

  const handleCollectReward = () => {
    if (subStep !== 'REWARD_AVAILABLE') return;
    currentTrialMetrics.current.interactionCount += 1;
    setScore(s => s + 100);
    finalizeTrialData(phase, trial, currentTrialMetrics.current.actualDuration);
    completeTrial();
  };

  const completeTrial = () => {
    trialActiveRef.current = false;
    setSubStep('IDLE');
    setIsAnchored(false);
    
    const maxTrials = phase === 'ACQUISITION' ? TOTAL_ACQ_TRIALS : EXTINCTION_TRIALS;
    if (trial + 1 < maxTrials) {
        setTrial(prev => prev + 1);
    } else {
        setPhase(phase === 'ACQUISITION' ? 'PIT_PREP' : 'RESULTS');
        setTrial(0);
    }
  };

  const startPIT = () => {
    setPhase('PIT');
    setChargeLevel(0);
    startTracking();
    const interval = setInterval(() => setShowCue(prev => !prev), 4000);
    setTimeout(() => {
      clearInterval(interval);
      finalizeTrialData('PIT_TRANSFER', 0, 30000);
      setPhase('EXT_PREP');
    }, PIT_DURATION);
  };

  const exportCSV = () => {
    if (sessionData.length === 0) return;
    const headers = Object.keys(sessionData[0]).join(',');
    const rows = sessionData.map(row => Object.values(row).join(','));
    const csvContent = "data:text/csv;charset=utf-8," + [headers, ...rows].join('\n');
    const encodedUri = encodeURI(csvContent);
    const link = document.createElement("a");
    link.setAttribute("href", encodedUri);
    link.setAttribute("download", "pavca_spatial_data.csv");
    document.body.appendChild(link);
    link.click();
    document.body.removeChild(link);
  };

  const PowerCell = ({ fastBlink = false }) => (
    <div className={`relative w-28 h-40 bg-gradient-to-b from-blue-800 to-blue-950 border-4 border-blue-400 rounded-xl shadow-[0_0_35px_rgba(96,165,250,0.6)] flex items-center justify-center transition-all cursor-crosshair group ${fastBlink ? 'animate-[pulse_0.4s_infinite]' : 'animate-pulse'}`}>
      <div className="absolute top-3 w-20 h-2 bg-blue-400/50 rounded-full" />
      <Zap className="text-white w-14 h-14 drop-shadow-[0_0_15px_rgba(255,255,255,0.8)]" />
      <div className="absolute bottom-3 w-20 h-2 bg-blue-400/50 rounded-full" />
    </div>
  );

  const VendingSlot = ({ state, onCollect }) => {
    const isOpen = state === 'REWARD_AVAILABLE';
    return (
      <div 
          onClick={isOpen ? onCollect : undefined}
          className={`relative w-48 h-56 bg-slate-800 border-4 rounded-t-3xl shadow-2xl transition-all duration-300 z-[100] ${isOpen ? 'border-yellow-400 cursor-pointer scale-110 bg-slate-700 shadow-[0_0_60px_rgba(234,179,8,0.4)]' : 'border-slate-700 opacity-40'}`}
      >
        <div className={`w-full h-12 flex items-center justify-center border-b-4 transition-colors ${isOpen ? 'bg-yellow-600 border-yellow-700' : 'bg-slate-700 border-slate-800'}`}>
          <span className="text-xs text-white font-black tracking-widest uppercase">{isOpen ? 'CLICK TO COLLECT' : 'LOCKED'}</span>
        </div>
        <div className="relative h-40 w-full bg-slate-900 flex items-center justify-center overflow-hidden">
             {isOpen && (
                 <div className="flex flex-col items-center justify-center animate-in zoom-in duration-300">
                    <div className="w-24 h-24 bg-gradient-to-tr from-yellow-600 via-yellow-400 to-yellow-100 rounded-full shadow-[0_0_40px_rgba(251,191,36,0.8)] border-4 border-white animate-bounce flex items-center justify-center">
                        <Trophy size={40} className="text-yellow-900" />
                    </div>
                 </div>
             )}
             {!isOpen && (
                 <div className="w-16 h-16 bg-slate-950 rounded-full border-4 border-slate-800 flex items-center justify-center opacity-20">
                     <Target size={32} className="text-slate-600" />
                 </div>
             )}
        </div>
      </div>
    );
  };

  const PhaseOverlay = ({ title, icon: Icon, description, buttonText, onStart, secondaryDesc }) => (
    <div className="bg-slate-900 p-12 rounded-[2.5rem] border border-slate-700 shadow-2xl max-w-3xl mx-auto text-center border-b-[12px] border-b-blue-600 animate-in fade-in slide-in-from-bottom-8 duration-500">
        <div className="mb-8 flex justify-center">
            <div className="p-6 bg-blue-600/10 rounded-full border border-blue-500/30">
                <Icon size={56} className="text-blue-400" />
            </div>
        </div>
        <h2 className="text-5xl font-black mb-6 tracking-tight uppercase italic text-white">{title}</h2>
        <p className="text-slate-300 text-xl mb-6 font-medium leading-relaxed">{description}</p>
        <div className="bg-slate-950/60 p-6 rounded-2xl mb-10 border border-slate-800 text-left relative overflow-hidden">
            <div className="absolute top-0 left-0 w-1 h-full bg-blue-500" />
            <h4 className="text-blue-400 font-bold text-xs uppercase mb-3 tracking-widest flex items-center gap-2"><Activity size={14}/> Operational Directives</h4>
            <p className="text-slate-400 text-sm leading-relaxed">{secondaryDesc}</p>
        </div>
        <button onClick={onStart} className="w-full bg-blue-600 hover:bg-blue-500 text-white font-black py-6 rounded-2xl transition-all flex items-center justify-center gap-4 group text-2xl shadow-xl shadow-blue-900/40 active:scale-95">
          {buttonText} <Play size={28} className="fill-current group-hover:translate-x-2 transition-transform" />
        </button>
    </div>
  );

  return (
    <div className="min-h-screen bg-slate-950 text-slate-100 font-sans selection:bg-blue-500/30 overflow-x-hidden cursor-default">
      <div className="p-5 border-b border-slate-800 flex justify-between items-center bg-slate-900/90 backdrop-blur-2xl sticky top-0 z-50">
        <div className="flex items-center gap-4">
          <div className="bg-blue-600 p-2 rounded-xl shadow-lg shadow-blue-900/40">
            <ShieldAlert className="text-white" size={24} />
          </div>
          <div>
            <span className="font-black tracking-tighter text-2xl block leading-none">NEURO-SYNC</span>
            <span className="text-[11px] text-slate-500 font-bold uppercase tracking-[0.3em]">Coordination v5.5</span>
          </div>
        </div>
        <div className="flex items-center gap-10">
            <div className="flex flex-col items-end">
                <span className="text-[10px] text-slate-500 font-bold uppercase tracking-widest">Incentive Gain</span>
                <div className="flex items-center gap-2">
                    <Trophy className="w-5 h-5 text-yellow-500" />
                    <span className="font-mono font-black text-2xl text-yellow-500">{score.toLocaleString()}</span>
                </div>
            </div>
        </div>
      </div>

      <main className="max-w-6xl mx-auto p-12" ref={containerRef} onMouseMove={handleMouseMove}>
        
        {phase === 'INSTRUCTIONS' && (
          <PhaseOverlay 
            title="Baseline Initiation"
            icon={Target}
            description="Operational baseline assessment for Field Engineers."
            secondaryDesc="Observe the Power Cell. When the Cell reaches critical charge (fast blink), the Dispenser will prime. You must manually click the Dispenser to collect tokens and proceed."
            buttonText="AUTHORIZE PROTOCOL"
            onStart={() => setPhase('ACQ_PREP')}
          />
        )}

        {phase === 'ACQ_PREP' && (
          <PhaseOverlay 
            title="Phase 1: Observation"
            icon={Activity}
            description={`Coupled Cycle Analysis (${TOTAL_ACQ_TRIALS} Trials).`}
            secondaryDesc="The Power Cell (right) will activate for 8-12 seconds. It will blink rapidly for exactly 3 seconds before the Dispenser opens. Click the Dispenser to retrieve tokens."
            buttonText="START PROTOCOL"
            onStart={() => setPhase('ACQUISITION')}
          />
        )}

        {(phase === 'ACQUISITION' || phase === 'EXTINCTION') && (
          <div className="relative h-[700px] w-full border-4 border-slate-800 rounded-[4rem] overflow-hidden bg-slate-900/20 shadow-2xl">
            <div className="absolute inset-0 bg-[radial-gradient(circle_at_center,_rgba(30,58,138,0.1),_transparent)] pointer-events-none" />
            
            <div className="absolute top-8 left-1/2 -translate-x-1/2 flex items-center gap-6 bg-slate-900 px-8 py-3 rounded-full border border-slate-700 shadow-2xl z-10">
                <span className={`font-black text-sm tracking-widest uppercase ${phase === 'EXTINCTION' ? 'text-red-400' : 'text-blue-400'}`}>
                    {phase}
                </span>
                <div className="w-1.5 h-1.5 rounded-full bg-slate-600" />
                <span className="text-slate-400 font-mono font-bold uppercase tracking-widest">Trial {trial + 1} of {phase === 'ACQUISITION' ? TOTAL_ACQ_TRIALS : EXTINCTION_TRIALS}</span>
            </div>

            {/* Reward Goal - Moved to Far Left */}
            <div className="absolute top-1/2 -translate-y-1/2 left-12">
                <VendingSlot state={subStep} onCollect={handleCollectReward} />
            </div>

            {/* Sign Cue - Moved to Far Right */}
            <div className="absolute top-1/2 -translate-y-1/2 right-12">
                {subStep === 'CUE' ? (
                  <PowerCell fastBlink={isFastBlinking} />
                ) : (
                  <div className={`w-28 h-40 border-4 rounded-xl border-dashed flex items-center justify-center transition-colors ${subStep === 'REWARD_AVAILABLE' ? 'border-yellow-500/50' : 'border-slate-800/40'}`}>
                     <ZapOff className={subStep === 'REWARD_AVAILABLE' ? 'text-yellow-600/50' : 'text-slate-800'} size={32} />
                  </div>
                )}
            </div>

            <div className="absolute bottom-12 left-1/2 -translate-x-1/2 flex flex-col items-center">
                <div className={`w-24 h-24 rounded-full border-4 transition-all duration-300 flex items-center justify-center ${isAnchored ? 'border-green-500 bg-green-500/20 shadow-[0_0_30px_rgba(34,197,94,0.4)]' : 'border-slate-800 bg-slate-950/50'}`}>
                    {isAnchored ? <CheckCircle2 className="text-green-500" size={40} /> : <MousePointer2 className="text-slate-700 animate-bounce" size={40} />}
                </div>
            </div>
          </div>
        )}

        {phase === 'PIT_PREP' && (
            <PhaseOverlay 
                title="Phase 2: Transfer"
                icon={Zap}
                description="Manual Vigor Assessment."
                secondaryDesc="Charge the system by clicking rapidly. Observe if the visual cues impact your manual energy output."
                buttonText="START TRANSFER"
                onStart={startPIT}
            />
        )}

        {phase === 'PIT' && (
            <div className="flex flex-col items-center justify-center h-[650px] relative bg-slate-900/30 rounded-[4rem] border-4 border-slate-800">
                <div className="text-center mb-16">
                    <h3 className="text-2xl font-black text-blue-400 mb-4 uppercase italic">Manual Output Meter</h3>
                    <div className="w-[500px] bg-slate-800 h-8 rounded-full border-4 border-slate-700 overflow-hidden">
                        <div className="h-full bg-blue-500 transition-all duration-150" style={{ width: `${chargeLevel}%` }} />
                    </div>
                </div>
                
                <button 
                    onMouseDown={() => {
                        setChargeLevel(prev => Math.min(100, prev + 8));
                        setScore(s => s + 5);
                        currentTrialMetrics.current.interactionCount = (currentTrialMetrics.current.interactionCount || 0) + 1;
                    }}
                    className="w-64 h-64 bg-slate-900 border-[10px] border-slate-800 rounded-full flex flex-col items-center justify-center active:scale-95 transition-all shadow-2xl hover:border-blue-500 group relative"
                >
                    <Zap className="text-blue-500 group-active:text-white transition-colors mb-2" size={64} />
                    <span className="font-black text-4xl text-slate-500 group-active:text-white uppercase">Charge</span>
                </button>

                {showCue && (
                    <div className="absolute right-12 top-1/2 -translate-y-1/2 scale-110 drop-shadow-2xl animate-in zoom-in duration-300">
                         <PowerCell fastBlink={false} />
                    </div>
                )}
            </div>
        )}

        {phase === 'EXT_PREP' && (
             <PhaseOverlay 
                title="Phase 3: Retention"
                icon={ZapOff}
                description="Final Persistence Calibration."
                secondaryDesc="Return to observation. Cues will appear but rewards have been exhausted."
                buttonText="FINALIZE"
                onStart={() => setPhase('EXTINCTION')}
            />
        )}

        {phase === 'RESULTS' && (
          <div className="bg-slate-900 p-12 rounded-[3.5rem] border border-slate-800 shadow-2xl animate-in zoom-in duration-500">
            <div className="flex items-center justify-between mb-12 pb-8 border-b border-slate-800">
                <h2 className="text-5xl font-black text-white italic uppercase tracking-tighter">Diagnostic Summary</h2>
                <button onClick={exportCSV} className="bg-blue-600 hover:bg-blue-500 text-white px-10 py-4 rounded-2xl flex items-center gap-4 font-black text-lg">
                  <Download size={24} /> EXPORT CSV
                </button>
            </div>
            
            <div className="bg-slate-950 rounded-[2rem] border border-slate-800 overflow-hidden shadow-inner max-h-[400px] overflow-y-auto">
                <table className="w-full text-left border-collapse">
                    <thead className="bg-slate-900/80 text-slate-500 sticky top-0 z-10 backdrop-blur-md">
                        <tr>
                            <th className="py-6 px-6 text-[11px] font-black uppercase">Phase/Trial</th>
                            <th className="py-6 px-4 text-[11px] font-black uppercase">Dur</th>
                            <th className="py-6 px-4 text-[11px] font-black uppercase">Latency</th>
                            <th className="py-6 px-4 text-[11px] font-black uppercase">Vigor</th>
                            <th className="py-6 px-4 text-[11px] font-black uppercase">Target</th>
                            <th className="py-6 px-4 text-[11px] font-black uppercase text-right">Sign Bias</th>
                        </tr>
                    </thead>
                    <tbody className="divide-y divide-slate-800 font-mono text-xs">
                        {sessionData.map((d, i) => (
                            <tr key={i} className="hover:bg-blue-600/5 transition-colors">
                                <td className="py-5 px-6">
                                    <div className="flex flex-col">
                                        <span className="text-slate-300 font-bold uppercase text-[10px]">{d.phase}</span>
                                        <span className="text-slate-600">Trial #{d.trial}</span>
                                    </div>
                                </td>
                                <td className="py-5 px-4 text-slate-400 font-bold">{d.duration}</td>
                                <td className="py-5 px-4 text-blue-400 font-bold">{d.latency}</td>
                                <td className="py-5 px-4 text-emerald-400 font-bold">{d.vigor}</td>
                                <td className="py-5 px-4">
                                    <span className={`px-3 py-1 rounded-full text-[10px] font-black ${d.target === 'CUE' ? 'bg-blue-900 text-blue-400' : d.target === 'GOAL' ? 'bg-yellow-900 text-yellow-400' : 'bg-slate-800 text-slate-500'}`}>
                                        {d.target}
                                    </span>
                                </td>
                                <td className="py-5 px-4 text-right text-slate-300 font-bold">{d.signBias}</td>
                            </tr>
                        ))}
                    </tbody>
                </table>
            </div>
          </div>
        )}
      </main>

      <footer className="fixed bottom-0 w-full p-4 bg-slate-950/80 border-t border-slate-900 flex justify-between px-12">
          <div className="flex items-center gap-6">
              <div className="flex items-center gap-3">
                <div className={`w-2.5 h-2.5 rounded-full ${subStep !== 'IDLE' ? 'bg-blue-500 animate-pulse' : 'bg-slate-700'}`} />
                <span className="text-[10px] text-slate-500 font-black uppercase">Telemetry Status: {subStep}</span>
              </div>
          </div>
          <div className="text-[10px] text-slate-700 font-black tracking-[0.3em] uppercase">NEURO-SYNC LABS</div>
      </footer>
    </div>
  );
};

export default App;
